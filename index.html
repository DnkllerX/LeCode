function solve(graph, n) { const dist = Array(n).fill(Infinity); dist[0] = 0; const pq = [[0, 0]]; while (pq.length) { const [d, u] = pq.shift(); for (const [v, w] of graph[u]) { if (dist[u] + w < dist[v]) { dist[v] = dist[u] + w; pq.push([dist[v], v]); } } } return dist; }

def knapsack(W, wt, val, n): dp = [[0]*(W+1) for _ in range(n+1)] for i in range(1, n+1): for w in range(W+1): if wt[i-1] <= w: dp[i][w] = max( val[i-1]+dp[i-1][w-wt[i-1]], dp[i-1][w] ) else: dp[i][w] = dp[i-1][w] return dp[n][W]
